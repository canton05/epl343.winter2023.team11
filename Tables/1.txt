CREATE TABLE  "APPLICATION_SOFTWARE" 
   (	"APPLICATION_SOFTWARE_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"APPLICATION_SOFTWARE_NAME" VARCHAR2(200) NOT NULL ENABLE, 
	"APPLICATION_SOFTWARE_PURPOSE" VARCHAR2(2000) NOT NULL ENABLE, 
	"APPLICATION_SOFTWARE_SCOPE" VARCHAR2(2000) NOT NULL ENABLE, 
	"APPLICATION_SOFTWARE_RESPONSIPLE" VARCHAR2(10) NOT NULL ENABLE, 
	"APPLICATION_SOFTWARE_CLIENT_CONTACT" VARCHAR2(45), 
	"APPLICATION_SOFTWARE_ADDRESSED_TO" VARCHAR2(2000), 
	"APPLICATION_SOFTWARE_CLIENT_ID" NUMBER, 
	 PRIMARY KEY ("APPLICATION_SOFTWARE_ID")
  USING INDEX  ENABLE
   )
/
